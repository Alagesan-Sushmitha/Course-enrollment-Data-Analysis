df["Term Without Session"] = df["Term"].str[:-1]
# Create a new DataFrame with the course and term as the index, and the count of students as the value
course_term_count = df.groupby(['Course Prefix', 'Term Without Session'])['Student'].nunique().unstack()

# Create a bar chart with each bar divided into three colors representing each term
course_term_count.plot(kind='bar', stacked=True, color=['#CFC7D2', '#BEA8AA', '#9E9885'])
plt.title('Course Participation by Term')
plt.xlabel('Course')
plt.ylabel('Number of Students')
plt.show()

Course Performance Analysis¶

In [3]:
course_means = df.groupby("Course Prefix")["Grade"].mean()

# Create bar chart showing average grade for each course
course_means.plot(kind="bar", x="Course Prefix", y="Grade")

# Create three line graphs (one for each term) showing average grade for each course in that term
for term in ["2201", "2204", "2207"]:
    term_data = df[df["Term Without Session"] == term]
    term_means = term_data.groupby("Course Prefix")["Grade"].mean()
    term_means.plot(label=term, linestyle="-", marker="o")

# Adjust appearance of chart
plt.title("Average Grade vs Courses")
plt.xlabel("Course")
plt.ylabel("Average Grade")
plt.legend()
plt.show()
